#include <bits/stdc++.h>
using namespace std;
const int N = 1e5 + 7;
int main() {
    int n;
    cin >> n;
    int a[N];
    for(int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    int res = 1;
    int dem = 1;

    for (int i = 2; i <= n; i++) {
        if (a[i] >= a[i-1]) {
            dem++;
            res = max(res, dem);
        } else {
            dem = 1;
        }
    }

    cout << res << endl;

    return 0;
}

#include <bits/stdc++.h>
using namespace std;

const int N = 10e5 + 7;
int l[N];

struct a {
    int bd, kt;
};

bool x(a u, a v) {
    return u.kt < v.kt;
}

int main() {
    int n;
    cin >> n;

    a a[n + 1];
    for (int i = 1; i <= n; i++) {
        cin >> a[i].bd;
        cin >> a[i].kt;
    }

    int ans = 0;
    sort(a + 1, a + n + 1, x);

    for (int i = 1; i <= n; i++) {
        l[i] = 1;
        for (int j = 1; j < i; j++) { 
            if (a[j].kt <= a[i].bd) {
                l[i] = max(l[i], l[j] + 1);
            }
            ans = max(ans, l[i]);
        }
    }

    cout << ans;
}

#include <iostream>
using namespace std;
const int N=10e5+7;
long long a[N];
bool f[N];
int main(){
    int n;
    cin>>n;
    int s=0;

    for(int i=1;i<=n;i++){
        cin>>a[i];
        s+=a[i];
    }
    f[0]=1;
    for(int i=1;i<=n;i++){
        for(int j=s;j>=a[i];j--){
            if(f[j]==0 && f[j-a[i]]==1){
                f[j]=1;
            }
        }
    }
    int dem=0;
    for(int i=1;i<=s;i++){
        if(f[i]==true){
            dem++;
        }
    }
    cout<<dem;
}
