#include <bits/stdc++.h>
using namespace std;
const int N = 1e5 + 7;
int main() {
    int n;
    cin >> n;
    int a[N];
    for(int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    int res = 1;
    int dem = 1;

    for (int i = 2; i <= n; i++) {
        if (a[i] >= a[i-1]) {
            dem++;
            res = max(res, dem);
        } else {
            dem = 1;
        }
    }

    cout << res << endl;

    return 0;

#include <bits/stdc++.h>
using namespace std;

const int N = 10e5 + 7;
int l[N];

struct a {
    int bd, kt;
};

bool x(a u, a v) {
    return u.kt < v.kt;
}

int main() {
    int n;
    cin >> n;

    a a[n + 1];
    for (int i = 1; i <= n; i++) {
        cin >> a[i].bd;
        cin >> a[i].kt;
    }

    int ans = 0;
    sort(a + 1, a + n + 1, x);

    for (int i = 1; i <= n; i++) {
        l[i] = 1;
        for (int j = 1; j < i; j++) { 
            if (a[j].kt <= a[i].bd) {
                l[i] = max(l[i], l[j] + 1);
            }
            ans = max(ans, l[i]);
        }
    }

    cout << ans;
}

#include<bits/stdc++.h>
using namespace std;
const int N=1e5+3;
int a[N], f[N], b[N], ans, res, giua,n;
int tknp (int dau, int cuoi, int x)
{
    ans=0;
    while (dau<=cuoi)
    {
        giua=(dau+cuoi)/2;
        if (a[b[giua]]<x)
        {
            ans=giua;
            dau=giua+1;
        }
        else
            cuoi=giua -1 ;
    }
    return b[ans];
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>n;
    for (int i=1; i<=n; i++)
    {
        cin>>a[i];
    }
    res=1;
    f[1]=1;
    b[1]=1;
    for (int i=2; i<=n; i++)
    {
        int jmax=tknp(1, f[res], a[i]); 
        f[i]=f[jmax]+1;
        b[f[i]]=i;
        if (f[i]>f[res]) res=i;
    }
    cout<<f[res];
    return 0;
}
